# v1.5.0 Implementation Plan — Cloud live-runtime test scaffolding + Cedar depth

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development. Steps use checkbox (`- [ ]`) syntax.

**Goal:** Close the last three deferred test-depth items: H8 (Cedar live engine integration beyond decision-parity), H18 (AWS Bedrock AgentCore live-runtime tests with real creds), H19 (GCP Vertex Agent Engine live-runtime tests with real creds). H18/H19 ship as a cred-gated framework that skips cleanly when credentials are absent — same shape as the existing `EAP_ENABLE_REAL_RUNTIMES=1` gate used by the unit tests, but actually exercises live cloud services when both the env flag and real credentials are present.

**Architecture:** New `cloud_live` pytest marker (orthogonal to the existing `cloud` marker for construction-only tests). Each live test is gated by:
1. The marker (`@pytest.mark.cloud_live`)
2. A pytest fixture that checks env var (`EAP_LIVE_AWS=1` / `EAP_LIVE_GCP=1`) and skips if absent
3. A try-import-and-call probe that skips with a clear "creds invalid or expired" message if the cred chain fails

The default gauntlet (`pytest -m "not extras and not cloud and not cloud_live"`) skips them all — CI without cred provisioning sees zero impact. Users with creds run them locally via `EAP_LIVE_AWS=1 pytest -m cloud_live` (or both flags for full coverage).

**Cedar depth (H8) is unrelated to cred gating** — `cedarpy` is just a library. T1 adds entity-store / slot / schema-validation tests to the existing `CedarPolicyEvaluator` test suite without any cloud dependency.

**Tech Stack:** Python 3.11+, `cedarpy>=4` (T1), `boto3>=1.34` (T2 — already in `[aws]` extra), `google-cloud-aiplatform>=1.50` (T3 — already in `[gcp]` extra). No new optional extras.

**Scope:**
- IN: Cedar entity stores + slot variables + schema validation; cloud-live test framework with `cloud_live` marker + env-flag + cred-probe skip discipline; ~5-6 AWS AgentCore smoke tests covering Memory + Registry; ~5-6 GCP Vertex smoke tests covering Memory Bank + Agent Registry.
- OUT (future minors): exhaustive coverage of every cloud service (Payments, Gateway, Code Sandbox, Browser Sandbox, Eval) — v1.5 establishes the framework and a representative cross-section; later minors add depth.

- [ ] **Step 2.2: Cloud-live framework — `conftest.py`**

```python
"""Fixtures for cloud-live integration tests.

Skips cleanly when env flags are missing or credentials fail a probe.
Sets EAP_ENABLE_REAL_RUNTIMES=1 for the duration so the SDK's
real-runtime gate opens.

Two-level gating:
1. ``EAP_LIVE_AWS=1`` / ``EAP_LIVE_GCP=1`` — opt-in env flag.
2. Cred probe — try a small no-op call (e.g. ``boto3.client('sts').get_caller_identity()``).
   If it fails, skip with a clear message naming the cred chain that needs
   attention (AWS_PROFILE, AWS_ACCESS_KEY_ID, etc.).
"""

from __future__ import annotations

import os
from typing import Any

import pytest


@pytest.fixture(scope="session")
def live_aws_enabled() -> None:
    """Skip the test if EAP_LIVE_AWS=1 is not set or AWS creds are invalid."""
    if os.environ.get("EAP_LIVE_AWS") != "1":
        pytest.skip(
            "Set EAP_LIVE_AWS=1 and provide AWS credentials "
            "(AWS_PROFILE / AWS_ACCESS_KEY_ID / etc.) to run cloud-live AWS tests."
        )
    try:
        import boto3
        sts = boto3.client("sts", region_name=os.environ.get("AWS_REGION", "us-east-1"))
        sts.get_caller_identity()
    except Exception as e:
        pytest.skip(f"AWS credentials invalid or unreachable: {e}")
    # Open the SDK's real-runtime gate for the test duration.
    os.environ["EAP_ENABLE_REAL_RUNTIMES"] = "1"


@pytest.fixture(scope="session")
def live_gcp_enabled() -> None:
    """Skip the test if EAP_LIVE_GCP=1 is not set or GCP creds are invalid."""
    if os.environ.get("EAP_LIVE_GCP") != "1":
        pytest.skip(
            "Set EAP_LIVE_GCP=1 and provide GCP credentials "
            "(GOOGLE_APPLICATION_CREDENTIALS or default ADC) to run cloud-live GCP tests."
        )
    try:
        import google.auth  # type: ignore[import-untyped]
        creds, _project = google.auth.default(
            scopes=["https://www.googleapis.com/auth/cloud-platform"]
        )
    except Exception as e:
        pytest.skip(f"GCP credentials invalid or unreachable: {e}")
    os.environ["EAP_ENABLE_REAL_RUNTIMES"] = "1"


@pytest.fixture(scope="session")
def aws_region() -> str:
    return os.environ.get("AWS_REGION", "us-east-1")


@pytest.fixture(scope="session")
def gcp_project_id() -> str:
    val = os.environ.get("GCP_PROJECT_ID") or os.environ.get("GOOGLE_CLOUD_PROJECT")
    if not val:
        pytest.skip("Set GCP_PROJECT_ID (or GOOGLE_CLOUD_PROJECT) to run GCP live tests.")
    return val


@pytest.fixture(scope="session")
def unique_test_id() -> str:
    """Per-test-session unique id for tagging cloud artifacts. Keeps
    test runs from colliding when multiple devs run concurrently."""
    import uuid
    return f"eap-live-{uuid.uuid4().hex[:12]}"
```

- [ ] **Step 2.3: Memory smoke tests** — `test_agentcore_memory_live.py`

Cover the AgentCore Memory contract end-to-end:

```python
"""Live AWS Bedrock AgentCore Memory integration tests.

Skipped unless EAP_LIVE_AWS=1 + valid AWS creds + AGENTCORE_MEMORY_ID
env var pointing at a test memory bank. Creates records tagged with
unique_test_id; cleans up via teardown.
"""

import os
import pytest

pytest.importorskip("boto3")
pytestmark = pytest.mark.cloud_live


@pytest.fixture
def agentcore_memory_id() -> str:
    val = os.environ.get("AGENTCORE_MEMORY_ID")
    if not val:
        pytest.skip("Set AGENTCORE_MEMORY_ID to an existing AgentCore Memory bank.")
    return val


async def test_remember_recall_roundtrip(
    live_aws_enabled, aws_region, agentcore_memory_id, unique_test_id
):
    """Round-trip: remember a fact, recall it, assert equality."""
    from eap_core.integrations.agentcore import AgentCoreMemoryStore

    store = AgentCoreMemoryStore(
        memory_id=agentcore_memory_id,
        region=aws_region,
    )
    session_id = f"sess-{unique_test_id}"
    key = "favorite_color"
    value = "vermillion"

    try:
        await store.remember(session_id, key, value)
        recalled = await store.recall(session_id, key)
        assert recalled == value
    finally:
        # Cleanup — best-effort.
        try:
            await store.clear(session_id)
        except Exception:
            pass


async def test_recall_missing_key_returns_none(
    live_aws_enabled, aws_region, agentcore_memory_id, unique_test_id
):
    """Recall against an absent key returns None (not an exception)."""
    from eap_core.integrations.agentcore import AgentCoreMemoryStore

    store = AgentCoreMemoryStore(
        memory_id=agentcore_memory_id,
        region=aws_region,
    )
    session_id = f"sess-{unique_test_id}-empty"
    result = await store.recall(session_id, "nonexistent")
    assert result is None


async def test_list_keys_returns_remembered_keys(
    live_aws_enabled, aws_region, agentcore_memory_id, unique_test_id
):
    """After remembering three keys, list_keys returns them all."""
    from eap_core.integrations.agentcore import AgentCoreMemoryStore

    store = AgentCoreMemoryStore(memory_id=agentcore_memory_id, region=aws_region)
    session_id = f"sess-{unique_test_id}-listkeys"
    try:
        await store.remember(session_id, "k1", "v1")
        await store.remember(session_id, "k2", "v2")
        await store.remember(session_id, "k3", "v3")
        keys = sorted(await store.list_keys(session_id))
        assert keys == ["k1", "k2", "k3"]
    finally:
        try:
            await store.clear(session_id)
        except Exception:
            pass
```
